pipeline {
    agent any

    stages {  // Add stages block
        stage('Build') {
            steps {
                sh '''
                python3 -m venv venv
                source venv/bin/activate
                pip install --upgrade pip
                pip install -r requirements.txt
                '''
            }
        }

        stage('Test') {
            steps {
                sh 'pytest'
            }
        }

        stage('Deploy') {
            steps {
                sh '''
                source venv/bin/activate
                nohup python app.py > app.log 2>&1 &
                '''
            }
        }
    } // Closing for `stages`
} // Closing for `pipeline`
