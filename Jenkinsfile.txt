pipeline {
    agent any

    stage('Build') {
        steps {
            sh '''
            python3 -m venv venv
            source venv/bin/activate
            pip install --upgrade pip
            pip install -r requirements.txt
            '''
            }
        }

        stage('Test') {
            steps {
                sh 'pytest'
            }
        }
        stage('Deploy') {
            steps {
                sh 'python app.py &'
            }
        }
    }
}
